\begin{align}
  p(y|\alpha, \sigma) &= \mathrm{Weibull}(y|\alpha, \sigma) \nonumber \\ 
  &= \frac{\alpha}{\sigma} \left(\frac{y}{\sigma}\right)^{\alpha - 1} \exp\left(-\left(\frac{y}{\sigma}\right)^{\alpha}\right)
  \label{eq:weibull}
\end{align}
